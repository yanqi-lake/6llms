#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    vector<int> a;
    int x;
    while (cin >> x) {
        a.push_back(x);
    }
    if (a.empty()) {
        cout << 0 << "\n" << 0 << "\n";
        return 0;
    }

    vector<int> tail1;
    tail1.reserve(a.size());
    for (int v : a) {
        int y = -v;
        auto it = upper_bound(tail1.begin(), tail1.end(), y);
        if (it == tail1.end()) tail1.push_back(y);
        else *it = y;
    }
    int lnnis_len = (int)tail1.size();

    vector<int> tail2;
    tail2.reserve(a.size());
    for (int v : a) {
        auto it = lower_bound(tail2.begin(), tail2.end(), v);
        if (it == tail2.end()) tail2.push_back(v);
        else *it = v;
    }
    int lis_len = (int)tail2.size();

    cout << lnnis_len << "\n" << lis_len << "\n";
    return 0;
}
