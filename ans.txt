#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

string highPrecisionAdd(string a, string b) {
    while (a.length() > 1 && a[0] == '0') a.erase(a.begin());
    while (b.length() > 1 && b[0] == '0') b.erase(b.begin());

    if (a.length() < b.length()) swap(a, b);

    b.insert(0, a.length() - b.length(), '0');

    string result;
    int carry = 0;

    for (int i = a.length() - 1; i >= 0; --i) {
        int sum = (a[i] - '0') + (b[i] - '0') + carry;
        result.push_back(sum % 10 + '0');
        carry = sum / 10;
    }

    if (carry > 0) result.push_back(carry + '0');

    reverse(result.begin(), result.end());

    while (result.length() > 1 && result[0] == '0') result.erase(result.begin());

    return result;
}

int main() {
    string a, b;
    cin >> a >> b;
    cout << highPrecisionAdd(a, b) << endl;
    return 0;
}
