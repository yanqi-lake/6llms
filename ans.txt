#include <bits/stdc++.h>
using namespace std;
using int64 = long long;
using i128 = __int128_t;

int64 mod_mul(int64 a, int64 b, int64 mod) {
    return (i128)a * b % mod;
}

int64 mod_pow(int64 a, int64 e, int64 mod) {
    int64 res = 1 % mod;
    a %= mod;
    while (e > 0) {
        if (e & 1) res = mod_mul(res, a, mod);
        a = mod_mul(a, a, mod);
        e >>= 1;
    }
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int64 N, K, M, P;
    if (!(cin >> N >> K >> M >> P)) return 0;
    
    if (P == 1) {
        cout << 0 << "\n";
        return 0;
    }
    
    if (K == 1) {
        cout << (N % P) << "\n";
        return 0;
    }
    
    int64 powM = mod_pow(M % P, K - 2, P);
    int64 term1 = mod_mul(M % P, N % P, P);
    int64 sum = M % P * ((M + 1) % P) % P;
    sum = sum * ((P + 1) / 2) % P;
    int64 term2 = mod_mul((K - 1) % P, sum, P);
    int64 diff = (term1 - term2) % P;
    if (diff < 0) diff += P;
    int64 result = mod_mul(diff, powM, P);
    
    cout << result << "\n";
    return 0;
}
