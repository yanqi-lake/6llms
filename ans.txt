#include <bits/stdc++.h>
using namespace std;

bool canPair(const string& a, const string& b) {
    if (a == b) return true;
    if (a.size() != b.size()) return false;
    for (size_t i = 0; i < a.size(); i++) {
        if (a[i] != b[a.size() - 1 - i]) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    string S;
    if (!(cin >> S)) return 0;
    
    int n = S.size();
    int L = 0, R = n - 1;
    int pairs = 0;
    
    while (L < R) {
        bool found = false;
        for (int i = L; i < R; i++) {
            string SL = S.substr(L, i - L + 1);
            for (int j = R; j > i; j--) {
                string SR = S.substr(j, R - j + 1);
                if (canPair(SL, SR)) {
                    L = i + 1;
                    R = j - 1;
                    pairs++;
                    found = true;
                    break;
                }
            }
            if (found) break;
        }
        if (!found) {
            cout << "NO" << endl;
            return 0;
        }
    }
    
    int total = pairs * 2;
    if (L == R) {
        total++;
    }
    
    if (total <= 1) {
        cout << "NO" << endl;
        return 0;
    }
    
    cout << "YES" << endl;
    cout << total << endl;
    
    return 0;
}
