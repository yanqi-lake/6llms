#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

vector<long long> compute_prefix_sums(vector<int>& coords) {
    sort(coords.begin(), coords.end());
    vector<long long> prefix(coords.size() + 1, 0);
    for (int i = 0; i < coords.size(); ++i) {
        prefix[i+1] = prefix[i] + coords[i];
    }
    return prefix;
}

long long compute_sum(const vector<int>& coords, const vector<long long>& prefix, int x) {
    int k = lower_bound(coords.begin(), coords.end(), x) - coords.begin();
    long long sum = (long long)x * k - prefix[k];
    sum += (prefix.back() - prefix[k]) - (long long)x * (coords.size() - k);
    return sum;
}

vector<pair<int, int>> get_ranges(const vector<int>& coords, const vector<long long>& prefix, long long max_sum) {
    int n = coords.size();
    int left = coords[0] - (int)(max_sum / n + 1);
    int right = coords.back() + (int)(max_sum / n + 1);

    int min_x = right;
    int max_x = left;
    for (int x = left; x <= right; ++x) {
        long long s = compute_sum(coords, prefix, x);
        if (s <= max_sum) {
            min_x = min(min_x, x);
            max_x = max(max_x, x);
        }
    }

    if (min_x > max_x) return {};
    return {{min_x, max_x}};
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, D;
    cin >> N >> D;
    vector<int> x_coords(N), y_coords(N);
    for (int i = 0; i < N; ++i) {
        cin >> x_coords[i] >> y_coords[i];
    }

    auto x_prefix = compute_prefix_sums(x_coords);
    auto y_prefix = compute_prefix_sums(y_coords);

    auto x_ranges = get_ranges(x_coords, x_prefix, D);
    if (x_ranges.empty()) {
        cout << "0\n";
        return 0;
    }

    long long result = 0;
    for (auto [x_min, x_max] : x_ranges) {
        for (int x = x_min; x <= x_max; ++x) {
            long long sx = compute_sum(x_coords, x_prefix, x);
            long long remaining = D - sx;
            if (remaining < 0) continue;

            auto y_ranges = get_ranges(y_coords, y_prefix, remaining);
            for (auto [y_min, y_max] : y_ranges) {
                result += y_max - y_min + 1;
            }
        }
    }

    cout << result << '\n';
    return 0;
}

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

using ll = long long;

ll S(const vector<int>& arr, const vector<ll>& pre, long long x) {
    int n = arr.size();
    int pos = lower_bound(arr.begin(), arr.end(), (int)x) - arr.begin();
    ll total = pre[n];
    return x * n - 2 * pre[pos] + total;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N;
    long long D;
    cin >> N >> D;
    
    vector<int> xs(N), ys(N);
    for (int i = 0; i < N; ++i) {
        cin >> xs[i] >> ys[i];
    }
    
    sort(xs.begin(), xs.end());
    sort(ys.begin(), ys.end());
    
    vector<ll> preX(N + 1, 0), preY(N + 1, 0);
    for (int i = 0; i < N; ++i) {
        preX[i + 1] = preX[i] + xs[i];
        preY[i + 1] = preY[i] + ys[i];
    }
    
    int xMid = xs[N / 2];
    int yMid = ys[N / 2];
    
    ll minSx = S(xs, preX, xMid);
    ll minSy = S(ys, preY, yMid);
    
    if (minSx + minSy > D) {
        cout << 0 << "\n";
        return 0;
    }
    
    long long xLow = (long long)xMid - (D - minSx);
    long long xHigh = (long long)xMid + (D - minSx);
    long long yLow = (long long)yMid - (D - minSy);
    long long yHigh = (long long)yMid + (D - minSy);
    
    long long yLeft = yMid;
    long long yRight = yMid;
    long long ans = 0;
    
    for (long long x = xLow; x <= xHigh; ++x) {
        ll sX = S(xs, preX, x);
        ll remaining = D - sX;
        
        if (remaining < minSy) continue;
        
        while (yLeft > yLow) {
            ll sLeft = S(ys, preY, yLeft - 1);
            if (sLeft <= remaining) {
                --yLeft;
            } else {
                break;
            }
        }
        
        while (yRight < yHigh) {
            ll sRight = S(ys, preY, yRight + 1);
            if (sRight <= remaining) {
                ++yRight;
            } else {
                break;
            }
        }
        
        ans += (yRight - yLeft + 1);
    }
    
    cout << ans << "\n";
    return 0;
}
