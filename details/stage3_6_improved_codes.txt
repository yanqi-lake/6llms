#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

typedef __int128 ll;

const int MAXN = 80;
const int MAXM = 80;

ll dp[MAXM][MAXM];
ll pow2[MAXM + 1];

string int128_to_string(ll x) {
    if (x == 0) return "0";
    string s;
    while (x > 0) {
        s.push_back('0' + x % 10);
        x /= 10;
    }
    reverse(s.begin(), s.end());
    return s;
}

ll solve_row(const vector<int>& row, int m) {
    for (int i = 0; i < m; ++i) {
        dp[i][i] = row[i] * pow2[m];
    }
    for (int len = 2; len <= m; ++len) {
        for (int l = 0; l <= m - len; ++l) {
            int r = l + len - 1;
            int t = m - len + 1;
            dp[l][r] = max(row[l] * pow2[t] + dp[l + 1][r], row[r] * pow2[t] + dp[l][r - 1]);
        }
    }
    return dp[0][m - 1];
}

int main() {
    int n, m;
    cin >> n >> m;
    vector<vector<int>> matrix(n, vector<int>(m));
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            cin >> matrix[i][j];
        }
    }

    pow2[0] = 1;
    for (int i = 1; i <= m; ++i) {
        pow2[i] = pow2[i - 1] * 2;
    }

    ll total_score = 0;
    for (int i = 0; i < n; ++i) {
        total_score += solve_row(matrix[i], m);
    }

    cout << int128_to_string(total_score) << endl;
    return 0;
}

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

typedef __int128 ll;

void print(__int128 x) {
    if (x == 0) {
        cout << "0" << endl;
        return;
    }
    if (x < 0) {
        cout << '-';
        x = -x;
    }
    string s;
    while (x > 0) {
        s += char('0' + x % 10);
        x /= 10;
    }
    reverse(s.begin(), s.end());
    cout << s << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<vector<int>> matrix(n, vector<int>(m));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> matrix[i][j];
        }
    }

    vector<ll> pow2(m + 1);
    pow2[0] = 1;
    for (int i = 1; i <= m; i++) {
        pow2[i] = pow2[i - 1] * 2;
    }

    ll totalScore = 0;

    for (int i = 0; i < n; i++) {
        vector<vector<ll>> dp(m, vector<ll>(m, 0));
        for (int len = 1; len <= m; len++) {
            for (int l = 0; l <= m - len; l++) {
                int r = l + len - 1;
                int round = m - len + 1; // round from 1 to m
                if (len == 1) {
                    dp[l][r] = (__int128)matrix[i][l] * pow2[round];
                } else {
                    ll takeLeft = (__int128)matrix[i][l] * pow2[round] + dp[l + 1][r];
                    ll takeRight = (__int128)matrix[i][r] * pow2[round] + dp[l][r - 1];
                    dp[l][r] = max(takeLeft, takeRight);
                }
            }
        }
        totalScore += dp[0][m - 1];
    }

    print(totalScore);

    return 0;
}
