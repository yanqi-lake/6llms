#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<vector<vector<long long>>> A(N+1, vector<vector<long long>>(N+1, vector<long long>(N+1)));
    for (int x = 1; x <= N; ++x) {
        for (int y = 1; y <= N; ++y) {
            for (int z = 1; z <= N; ++z) {
                cin >> A[x][y][z];
            }
        }
    }

    vector<vector<vector<long long>>> prefix(N+1, vector<vector<long long>>(N+1, vector<long long>(N+1, 0)));
    for (int x = 1; x <= N; ++x) {
        for (int y = 1; y <= N; ++y) {
            for (int z = 1; z <= N; ++z) {
                prefix[x][y][z] = A[x][y][z] 
                                + prefix[x-1][y][z] + prefix[x][y-1][z] + prefix[x][y][z-1]
                                - prefix[x-1][y-1][z] - prefix[x-1][y][z-1] - prefix[x][y-1][z-1]
                                + prefix[x-1][y-1][z-1];
            }
        }
    }

    int Q;
    cin >> Q;
    while (Q--) {
        int Lx, Rx, Ly, Ry, Lz, Rz;
        cin >> Lx >> Rx >> Ly >> Ry >> Lz >> Rz;

        long long sum = prefix[Rx][Ry][Rz]
                      - prefix[Lx-1][Ry][Rz] - prefix[Rx][Ly-1][Rz] - prefix[Rx][Ry][Lz-1]
                      + prefix[Lx-1][Ly-1][Rz] + prefix[Lx-1][Ry][Lz-1] + prefix[Rx][Ly-1][Lz-1]
                      - prefix[Lx-1][Ly-1][Lz-1];

        cout << sum << '\n';
    }

    return 0;
}

#include <iostream>

using namespace std;

long long S[105][105][105];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;

    for (int x = 1; x <= N; ++x) {
        for (int y = 1; y <= N; ++y) {
            for (int z = 1; z <= N; ++z) {
                long long val;
                cin >> val;
                S[x][y][z] = val
                    + S[x-1][y][z] + S[x][y-1][z] + S[x][y][z-1]
                    - S[x-1][y-1][z] - S[x-1][y][z-1] - S[x][y-1][z-1]
                    + S[x-1][y-1][z-1];
            }
        }
    }

    int Q;
    cin >> Q;
    while (Q--) {
        int lx, rx, ly, ry, lz, rz;
        cin >> lx >> rx >> ly >> ry >> lz >> rz;
        long long ans = S[rx][ry][rz]
            - S[lx-1][ry][rz] - S[rx][ly-1][rz] - S[rx][ry][lz-1]
            + S[lx-1][ly-1][rz] + S[lx-1][ry][lz-1] + S[rx][ly-1][lz-1]
            - S[lx-1][ly-1][lz-1];
        cout << ans << "\n";
    }

    return 0;
}
