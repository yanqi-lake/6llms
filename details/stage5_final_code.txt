#include <bits/stdc++.h>
using namespace std;

struct Carpet {
    long long a, b, g, k;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    if (!(cin >> n)) return 0;
    vector<Carpet> carpets(n);
    for (int i = 0; i < n; ++i) {
        long long a, b, g, k;
        cin >> a >> b >> g >> k;
        carpets[i] = {a, b, g, k};
    }
    long long x, y;
    cin >> x >> y;
    int ans = -1;
    for (int i = n - 1; i >= 0; --i) {
        const auto &c = carpets[i];
        if (x >= c.a && x <= c.a + c.g && y >= c.b && y <= c.b + c.k) {
            ans = i + 1;
            break;
        }
    }
    cout << ans;
    return 0;
}
