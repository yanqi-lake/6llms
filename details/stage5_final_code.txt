#include <bits/stdc++.h>
using namespace std;

long long mod_pow(long long a, long long e, long long mod) {
    long long res = 1 % mod;
    a %= mod;
    while (e > 0) {
        if (e & 1) res = (__int128)res * a % mod;
        a = (__int128)a * a % mod;
        e >>= 1;
    }
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    long long N, K, M, P;
    if (!(cin >> N >> K >> M >> P)) return 0;
    
    if (P == 1) {
        cout << 0 << "\n";
        return 0;
    }
    
    long long pow_mod = mod_pow(M % P, K - 1, P);
    
    __int128 A = K - 1;
    __int128 B = M + 1;
    __int128 AB = A * B;
    
    long long t_mod;
    if ((AB % 2) == 0) {
        t_mod = (long long)((AB / 2) % P);
    } else {
        if ((A % 2) == 0) {
            t_mod = (long long)(((A / 2) * B) % P);
        } else {
            t_mod = (long long)((A * (B / 2)) % P);
        }
    }
    
    long long n_mod = N % P;
    long long diff = n_mod - t_mod;
    diff %= P;
    if (diff < 0) diff += P;
    
    long long result = (__int128)pow_mod * diff % P;
    cout << result << "\n";
    
    return 0;
}
