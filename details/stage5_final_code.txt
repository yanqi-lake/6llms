#include <bits/stdc++.h>
using namespace std;

struct Carpet {int a,b,g,k;};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    if(!(cin>>n)) return 0;
    vector<Carpet> carpets;
    carpets.reserve(n);
    for(int i=0;i<n;i++){
        Carpet c;
        cin>>c.a>>c.b>>c.g>>c.k;
        carpets.push_back(c);
    }
    int x,y;
    cin>>x>>y;
    for(int i=n-1;i>=0;i--){
        const Carpet &c = carpets[i];
        if(x>=c.a && x<=c.a+c.g && y>=c.b && y<=c.b+c.k){
            cout<<i+1<<"\n";
            return 0;
        }
    }
    cout<<-1<<"\n";
    return 0;
}
