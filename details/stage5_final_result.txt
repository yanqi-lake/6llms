#include <iostream>
using namespace std;

typedef long long ll;

ll mul_mod(ll a, ll b, ll mod) {
    return (__int128)a * b % mod;
}

ll pow_mod(ll base, ll exp, ll mod) {
    ll result = 1;
    while (exp) {
        if (exp & 1) result = mul_mod(result, base, mod);
        base = mul_mod(base, base, mod);
        exp >>= 1;
    }
    return result;
}

int main() {
    ll N, K, M, P;
    cin >> N >> K >> M >> P;

    if (K == 1) {
        cout << N % P << endl;
        return 0;
    }

    ll mod = 2 * P;
    ll power = pow_mod(M, K - 1, mod);
    ll X = (2 * (N % mod) % mod - mul_mod(K - 1, M + 1, mod) + mod) % mod;
    ll result = mul_mod(power, X, mod);
    result = (result % mod + mod) % mod / 2;
    cout << result % P << endl;

    return 0;
}
