#include <iostream>
#include <vector>

using namespace std;

struct Carpet {
    int a, b, g, k;
};

int main() {
    int n;
    cin >> n;

    vector<Carpet> carpets(n);
    for (int i = 0; i < n; ++i) {
        cin >> carpets[i].a >> carpets[i].b >> carpets[i].g >> carpets[i].k;
    }

    int x, y;
    cin >> x >> y;

    for (int i = n - 1; i >= 0; --i) {
        if (carpets[i].a <= x && x <= carpets[i].a + carpets[i].g &&
            carpets[i].b <= y && y <= carpets[i].b + carpets[i].k) {
            cout << i + 1 << endl;
            return 0;
        }
    }

    cout << -1 << endl;
    return 0;
}

#include <bits/stdc++.h>

struct Carpet {int a,b,g,k;};
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    if(!(cin>>n)) return 0;
    vector<Carpet> carpets;
    carpets.reserve(n);
    for(int i=0;i<n;i++){
        Carpet c;
        cin>>c.a>>c.b>>c.g>>c.k;
        carpets.push_back(c);
    }
    int x,y;
    cin>>x>>y;
    for(int i=n-1;i>=0;i--){
        const Carpet &c = carpets[i];
        if(x>=c.a && x<=c.a+c.g && y>=c.b && y<=c.b+c.k){
            cout<<i+1<<"\n";
            return 0;
        }
    }
    cout<<-1<<"\n";
    return 0;
}

#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<vector<int>> carpets;
    for (int i = 0; i < n; i++) {
        int a, b, g, k;
        cin >> a >> b >> g >> k;
        carpets.push_back({a, b, g, k});
    }
    int x, y;
    cin >> x >> y;
    for (int i = n - 1; i >= 0; i--) {
        int a = carpets[i][0];
        int b = carpets[i][1];
        int g = carpets[i][2];
        int k = carpets[i][3];
        if (x >= a && x <= a + g && y >= b && y <= b + k) {
            cout << i + 1 << endl;
            return 0;
        }
    }
    cout << -1 << endl;
    return 0;
}
