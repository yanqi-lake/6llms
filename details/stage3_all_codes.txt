#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    
    int a[n], b[n], g[n], k[n];
    for (int i = 0; i < n; ++i) {
        cin >> a[i] >> b[i] >> g[i] >> k[i];
    }
    
    int x, y;
    cin >> x >> y;
    
    int result = -1;
    for (int i = n - 1; i >= 0; --i) {
        if (x >= a[i] && x <= a[i] + g[i] && y >= b[i] && y <= b[i] + k[i]) {
            result = i + 1;
            break;
        }
    }
    
    cout << result << endl;
    return 0;
}

#include <bits/stdc++.h>
using namespace std;

struct Carpet {
    long long a, b, g, k;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    if (!(cin >> n)) return 0;
    vector<Carpet> carpets(n);
    for (int i = 0; i < n; ++i) {
        long long a, b, g, k;
        cin >> a >> b >> g >> k;
        carpets[i] = {a, b, g, k};
    }
    long long x, y;
    cin >> x >> y;
    int ans = -1;
    for (int i = n - 1; i >= 0; --i) {
        const auto &c = carpets[i];
        if (x >= c.a && x <= c.a + c.g && y >= c.b && y <= c.b + c.k) {
            ans = i + 1;
            break;
        }
    }
    cout << ans;
    return 0;
}

#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<vector<int>> carpets;
    for (int i = 0; i < n; i++) {
        int a, b, g, k;
        cin >> a >> b >> g >> k;
        carpets.push_back({a, b, g, k});
    }
    int x, y;
    cin >> x >> y;
    for (int i = n - 1; i >= 0; i--) {
        int a = carpets[i][0];
        int b = carpets[i][1];
        int g = carpets[i][2];
        int k = carpets[i][3];
        if (x >= a && x <= a + g && y >= b && y <= b + k) {
            cout << i + 1 << endl;
            return 0;
        }
    }
    cout << -1 << endl;
    return 0;
}
